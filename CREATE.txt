CREATE IF NOT EXISTS Begeleider (
	begeleider_id SERIAL PRIMARY KEY NOT NULL,
	naam varchar(50) not null,
	telefoon int not null,
	adres varchar(45) not null,
	gemeente varchar(45) not null,
	email varchar(45) not null
);

CREATE IF NOT EXISTS Klant (
	klant_id SERIAL PRIMARY KEY NOT NULL,
	naam varchar(50) not null,
	telefoon int not null,
	betalingsgegevens varchar(50),
	adres varchar(45) not null,
	gemeente varchar(45) not null,
	email varchar(45) not null
);

CREATE IF NOT EXISTS Factuur (
	factuur_id SERIAL PRIMARY KEY NOT NULL,
	factuur_datum date not null,
	betaald boolean not null,
	programma varchar(45) not null,
	
);
